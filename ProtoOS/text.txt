x86_64-elf-gcc -ffreestanding -m32 -Wno-main -c kernel/src/kernel.c -o kernel/obj/kernel.o
x86_64-elf-ld -m elf_i386 -Ttext 0x1000 --oformat binary kernel/obj/kernel.o -o kernel/bin/kernel.bin

dd if=/dev/zero of=kernel/bin/kernel_padded.bin bs=512 count=1
dd if=kernel/bin/kernel.bin of=kernel/bin/kernel_padded.bin conv=notrunc

cat boot/bin/boot_kernel.bin kernel/bin/kernel_padded.bin > boot/bin/os.img

qemu-system-x86_64 boot/bin/os.img
